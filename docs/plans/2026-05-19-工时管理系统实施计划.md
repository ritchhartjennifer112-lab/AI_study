def split_simple(total_hours, work_orders, strategy="primary", primary=None, custom_hours=None):
    """简化模式工时切分"""
    n = len(work_orders)
    if n == 0:
        return {}

    if strategy == "average":
        per = total_hours / n
        return {wo: round(per, 2) for wo in work_orders}

    if strategy == "primary":
        if primary not in work_orders:
            primary = work_orders[0]
        others = [wo for wo in work_orders if wo != primary]
        result = {primary: round(total_hours * 0.6, 2)}
        if others:
            per_other = (total_hours * 0.4) / len(others)
            for wo in others:
                result[wo] = round(per_other, 2)
        return result

    if strategy == "custom":
        return custom_hours or {}

    return {}


def split_precise(periods):
    """精确模式：时间段并行切分"""
    from datetime import datetime

    def parse_time(t):
        return datetime.strptime(t, "%H:%M")

    result = {}
    for p in periods:
        start = parse_time(p["start"])
        end = parse_time(p["end"])
        duration = (end - start).total_seconds() / 3600
        orders = p["orders"]
        if not orders:
            continue
        per_order = duration / len(orders)
        for wo in orders:
            result[wo] = result.get(wo, 0) + per_order
    return {k: round(v, 2) for k, v in result.items()}
```

from core.constants import FAMILY_A_UNITS, COEFFICIENTS


def calculate_family_a(params):
    """计算族A（单机加工设备）标准工时"""
    total = 0
    breakdown = {}

    # A-01 落地调平
    a01 = FAMILY_A_UNITS["A-01"]["base"] * params.get("module_count", 1)
    breakdown["A-01"] = round(a01, 2)
    total += a01

    # A-02 导轨安装
    a02 = FAMILY_A_UNITS["A-02"]["base"] * params.get("rail_length_m", 1)
    breakdown["A-02"] = round(a02, 2)
    total += a02

    # A-03 丝杆组件
    a03 = FAMILY_A_UNITS["A-03"]["base"] * params.get("screw_set_count", 1)
    breakdown["A-03"] = round(a03, 2)
    total += a03

    # A-04 支架固定
    a04 = FAMILY_A_UNITS["A-04"]["base"] * params.get("bracket_count", 1)
    breakdown["A-04"] = round(a04, 2)
    total += a04

    # A-05 电机座
    a05 = FAMILY_A_UNITS["A-05"]["base"] * params.get("motor_base_count", 1)
    breakdown["A-05"] = round(a05, 2)
    total += a05

    # A-06 夹钳机构
    a06 = FAMILY_A_UNITS["A-06"]["base"] * params.get("clamp_module_count", 1)
    breakdown["A-06"] = round(a06, 2)
    total += a06

    # A-07 感应器
    a07 = FAMILY_A_UNITS["A-07"]["base"] * params.get("sensor_count", 1)
    breakdown["A-07"] = round(a07, 2)
    total += a07

    # A-08 整机调试
    a08 = FAMILY_A_UNITS["A-08"]["base"] + 0.5 * params.get("component_count", 0)
    breakdown["A-08"] = round(a08, 2)
    total += a08

    return {
        "total_hours": round(total, 2),
        "unit_breakdown": breakdown,
        "quality_level": "C",
    }


def apply_coefficients(base, coeffs):
    """应用修正系数"""
    result = base
    for cat, key in coeffs.items():
        result *= COEFFICIENTS[cat].get(key, 1.0)
    return round(result, 2)

def calculate_deviation(actual, standard):
    """计算偏差"""
    dev_hours = actual - standard
    dev_rate = dev_hours / standard if standard != 0 else 0
    status = "normal"
    if dev_rate > 0.30:
        status = "over"
    elif dev_rate < -0.30:
        status = "under"
    return {
        "actual_hours": actual,
        "standard_hours": standard,
        "deviation_hours": round(dev_hours, 2),
        "deviation_rate": round(dev_rate, 4),
        "status": status,
    }


def calculate_efficiency(actual, standard):
    """计算工时效率 = 标准 / 实际"""
    if actual == 0:
        return 0
    return round(standard / actual, 4)
```

import pandas as pd
import streamlit as st
from pathlib import Path


def import_from_excel(file_path):
    """从手动导出的 IN3 Excel 读取工单数据"""
    df = pd.read_excel(file_path)
    # 列名映射（根据实际 IN3 导出格式调整）
    column_map = {
        "工单号": "order_id",
        "设备型号": "device_model",
        "产品族": "family",
        "计划开工日": "plan_start",
        "计划完工日": "plan_end",
        "状态": "status",
    }
    df = df.rename(columns=column_map)
    df["plan_start"] = pd.to_datetime(df["plan_start"])
    df["plan_end"] = pd.to_datetime(df["plan_end"])
    return df


def get_active_orders(data_dir="data/erp_exports"):
    """获取最新的在制工单"""
    path = Path(data_dir)
    if not path.exists():
        return pd.DataFrame()
    # 读取目录下最新的 Excel
    files = sorted(path.glob("*.xlsx"), key=lambda x: x.stat().st_mtime, reverse=True)
    if not files:
        return pd.DataFrame()
    return import_from_excel(files[0])


def save_uploaded_file(uploaded_file, data_dir="data/erp_exports"):
    """保存用户上传的 IN3 导出文件"""
    path = Path(data_dir)
    path.mkdir(parents=True, exist_ok=True)
    save_path = path / uploaded_file.name
    with open(save_path, "wb") as f:
        f.write(uploaded_file.getbuffer())
    return save_path
```

            report_dir.mkdir(parents=True, exist_ok=True)
            report_file = report_dir / f"{date.strftime('%Y%m%d')}.csv"

            df_new = pd.DataFrame([{
                "date": record["date"],
                "employee": record["employee"],
                "order_id": wo,
                "hours": h,
                "is_primary": (wo == primary),
                "remark": remark,
            } for wo, h in custom_hours.items()])

            if report_file.exists():
                df_existing = pd.read_csv(report_file)
                # 去重：同一天同一员工同一工单只保留最新
                df_existing = df_existing[~((df_existing["date"] == record["date"]) &
                                            (df_existing["employee"] == record["employee"]) &
                                            (df_existing["order_id"].isin(selected_orders)))]
                df_combined = pd.concat([df_existing, df_new], ignore_index=True)
            else:
                df_combined = df_new

            df_combined.to_csv(report_file, index=False)
            st.success(f"✅ 已保存 {len(custom_hours)} 条记录")
```
