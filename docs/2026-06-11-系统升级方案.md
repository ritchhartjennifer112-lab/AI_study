| 机制 | MiMoCode | 本项目 | 评估 |

### 3.1 数据问题现状

### 3.2 数据清洗策略

```python
# core/data_quality_check.py

def check_data_quality():
    """数据质量检查：在构建技能矩阵前执行"""

    issues = []

    # 1. 检查工时数据一致性
    #    Excel累计值 vs 日报实际值
    csv_hours = get_csv_total_hours()
    db_hours = get_db_total_hours()
    if abs(csv_hours - db_hours) > 100:  # 阈值100h
        issues.append(f"工时数据不一致: CSV={csv_hours}, DB={db_hours}")

    # 2. 检查标准工时完整性
    #    族B/C无标准工时基准
    families = ['A', 'B', 'C']
    for f in families:
        if not has_std_time(f):
            issues.append(f"族{f}缺少标准工时基准")

    # 3. 检查工单计划时间
    #    plan_end全空
    empty_plan = count_empty_plan_end()
    if empty_plan > 0:
        issues.append(f"{empty_plan}个工单缺少计划完成时间")

    # 4. 检查员工数据完整性
    #    有效员工数
    active_employees = count_active_employees()
    if active_employees < 30:
        issues.append(f"有效员工数偏少: {active_employees}人")

    return issues
```

### 3.3 数据清洗规则

| 问题 | 清洗规则 | 处理方式 |
|------|----------|----------|
| 工时不一致 | 以日报CSV为准 | 标记差异，人工确认 |
| 标准工时缺失 | 族B/C暂用估算值 | 标注"估算"，置信度低 |
| 计划时间为空 | 从ERP同步 | Phase 2实现 |
| 员工数据不完整 | 只统计有报工记录的 | 排除无数据员工 |

```python
def build_skill_matrix_from_history():
    """从历史报工数据自动构建技能矩阵"""

    # 1. 关联：报工 → 工单 → 设备
    # 2. 按员工×设备聚合
    # 3. 计算综合评分（工单数×工时×天数）
    # 4. 按评分排序，划分效率等级
    # 5. 写入 skill_matrix 表

    # 效率等级计算逻辑：
    # - 综合评分 = 工单数 × 总工时 × 活跃天数
    # - 按设备分组，评分排名前30%为"快"，中间40%为"中"，后30%为"慢"
    # - 注意：这是相对效率（同设备内比较），不是绝对效率

    return {
        'employee': '张三',
        'device': '铜排仓',
        'total_hours': 120,
        'order_count': 5,
        'days_active': 20,
        'composite_score': 1200,  # 5 × 120 × 20
        'efficiency': '快'  # 同设备内排名前30%
    }
```

**效率等级定义**：
| 等级 | 定义 | 用途 |
|------|------|------|
| 快 | 同设备内综合评分前30% | 优先派工 |
| 中 | 同设备内综合评分中间40% | 正常派工 |
| 慢 | 同设备内综合评分后30% | 辅助岗位 |

**技能矩阵用途**：
| 用途 | 说明 |
|------|------|

**约束条件定义**：

```python
def filter_by_capacity(candidates):
    """过滤产能：排除已满负荷的员工"""

    # 当前产能计算逻辑：
    # - 月度标准工时 = 22天 × 10.5h/天 = 231h
    # - 已用工时 = 本月累计报工小时数
    # - 剩余产能 = 标准工时 - 已用工时
    # - 满负荷判定：剩余产能 < 20h

    result = []
    for emp in candidates:
        monthly_hours = get_monthly_hours(emp['name'])
        remaining = 231 - monthly_hours  # 标准月工时231h
        if remaining >= 20:  # 至少剩余20h
            emp['remaining_capacity'] = remaining
            result.append(emp)
    return result

def filter_by_overtime(candidates):
    """过滤加班：排除已超标的员工"""

    # 加班计算逻辑（按月汇总）：
    # - 标准日工时 = 10.5h
    # - 月度加班 = Σ(每日工时 - 10.5h)（仅计算超出部分）
    # - 加班阈值（按薪资等级）：
    #   - 高薪（大工）：36h/月
    #   - 中薪：48h/月
    #   - 普薪：60h/月

    result = []
    for emp in candidates:
        overtime = get_monthly_overtime(emp['name'])
        threshold = get_overtime_threshold(emp['salary_level'])
        if overtime < threshold:
            emp['remaining_overtime'] = threshold - overtime
            result.append(emp)
    return result
```

```python
# core/wps_sync.py

from playwright.sync_api import sync_playwright
import time

def download_from_wps(share_url, save_path):
    """从WPS共享文档下载Excel文件"""

    with sync_playwright() as p:
        # 1. 启动浏览器（有头模式，需登录）
        browser = p.chromium.launch(headless=False)
        page = browser.new_page()

        # 2. 打开共享链接
        page.goto(share_url)

        # 3. 等待登录（人工操作一次，保存cookie）
        input("请手动登录后按回车继续...")

        # 4. 找到下载按钮并点击
        download_btn = page.locator('button:has-text("下载")')

        # 5. 等待下载完成
        with page.expect_download() as download_info:
            download_btn.click()
        download = download_info.value
        download.save_as(save_path)

        browser.close()

    return save_path

# 定时任务配置
def setup_daily_sync():
    """配置每天凌晨自动同步"""
    import schedule

    schedule.every().day.at("02:00").do(
        download_from_wps,
        share_url="https://www.kdocs.cn/l/xxx",  # WPS共享链接
        save_path="data/production_plan_latest.xlsx"
    )
```
